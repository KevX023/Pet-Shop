import java.util.ArrayList;
import java.util.List;

public class PetShop {
    public static void main(String[] args) {
        // Create an ArrayList for dogs
        List<String> dogs = new ArrayList<>();
        dogs.add("Buddy");
        dogs.add("Max");
        dogs.add("Charlie");
        dogs.add("Bella");
        
        // Create an ArrayList for cats
        List<String> cats = new ArrayList<>();
        cats.add("Whiskers");
        cats.add("Luna");
        cats.add("Simba");
        cats.add("Cleo");
        
        // Print the dogs ArrayList
        System.out.println("Dogs:");
        for (String dog : dogs) {
            System.out.println(dog);
        }
        
        // Print the cats ArrayList
        System.out.println("\nCats:");
        for (String cat : cats) {
            System.out.println(cat);
        }
    }
}
